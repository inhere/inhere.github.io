git дҪҝз”ЁдёҖдәӣи®°еҪ•пјҢеёёз”Ёе‘Ҫд»Өе’ҢдёҖдәӣзү№ж®ҠйҖүйЎ№

pullе’ҢmergeвҖӢ

жҸҗеҖЎж·»еҠ йҖүйЎ№ --no-ff -- дёҚдҪҝз”Ёеҝ«йҖҹеҗҲе№¶

ж–Үд»¶жқғйҷҗдҝ®ж”№еј•иө·зҡ„еҶІзӘҒвҖӢ

йҖҡеёёжҳҜеӣ дёәејҖеҸ‘зҺҜеўғе’ҢиҝҗиЎҢзҺҜеўғжҳҜдёҚеҗҢзҡ„зі»з»ҹпјҢеҸ‘еёғйЎ№зӣ®еҲ°зәҝдёҠж—¶пјҢеҫҲеӨҡж—¶еҖҷйңҖиҰҒдҝ®ж”№ж–Үд»¶зҡ„жқғйҷҗгҖӮдҪҶgitжҠҠж–Үд»¶жқғйҷҗд№ҹеҲ—е…ҘдәҶзүҲжң¬з®ЎзҗҶгҖӮ

и§ЈеҶіеҠһжі•пјҡвҖӢ

gitдёӯеҸҜд»ҘеҠ е…ҘеҝҪз•Ҙж–Үд»¶жқғйҷҗзҡ„й…ҚзҪ®пјҢе…·дҪ“еҰӮдёӢпјҡ

$ git config core.filemode false // д»…еҜ№еҪ“еүҚйЎ№зӣ®жңүж•Ҳ
$ git config --global core.filemode false // е°Ҷй…ҚзҪ®еә”з”ЁеҲ°е…ЁеұҖ

иҝҷж ·е°ұи®ҫзҪ®дәҶеҝҪз•Ҙж–Үд»¶жқғйҷҗгҖӮжҹҘзңӢдёӢй…ҚзҪ®пјҡ

$ git config -l

alias.co=checkout
alias.br=branch
core.filemode=false
core.repositoryformatversion=0

Gitжү“TAGвҖӢ

жү“TAGд№ҹе°ұжҳҜеҸ‘еёғзүҲжң¬

$ git tag -a v1.2 -m "version 1.4"
$ git push --tags

Gitжү“еҢ…жәҗз ҒвҖӢ

еҜ№Gitз®ЎзҗҶзҡ„жәҗз ҒиҝӣиЎҢеҺӢзј©жү“еҢ…пјҢеҰӮжһңдҪҝз”Ёtar xvzf xxxx.tag.gz xxxxxзҡ„иҜқе№¶дёҚжҳҜдёҖдёӘеҫҲеҘҪзҡ„йҖүжӢ©пјҢеӣ дёәдјҡе°Ҷ.git/зӣ®еҪ•дёӢзҡ„дёӯй—ҙж–Үд»¶е…ЁйғЁеҺӢзј©пјҢеҰӮжһңеҸӘжғіиҰҒжҹҗдёҖдёӘзүҲжң¬зҡ„жәҗз ҒгҖӮ GitжҸҗдҫӣдәҶarchive е®ғдјҡз»ҷжү“еҢ…дёҖд»ҪзәҜеҮҖзҡ„д»Јз ҒгҖӮеҪ“з„¶иҝҷдёӘеҸӘйҖӮз”ЁдәҺеҸ‘йғЁдёҖдёӘзүҲжң¬зҡ„жәҗз ҒпјҢиҖҢдёҚжҳҜеӨҮд»ҪGitз®ЎзҗҶзҡ„ж•ҙеҘ—жәҗз ҒгҖӮ

$ git archive xxxx

xxxд»ЈиЎЁSHA-1 HashеҖјгҖӮ

GitеҸӘcloneжңҖж–°зүҲжң¬д»Јз ҒдёҚиҰҒ.gitзӣ®еҪ•вҖӢ

git clone --depth=1 git://someserver/somerepo dirformynewrepo
rm -rf !$/.git

дёҚдә§з”ҹ merge commitвҖӢ

йҖҡеёё merge/pull дјҡдә§з”ҹдёҖдёӘ merge commit,и®©жҸҗдәӨи®°еҪ•зңӢиө·жқҘдёҚеӨӘзҫҺи§ӮпјҢеҸҜд»ҘдҪҝз”Ё rebaseйҖүйЎ№жқҘеҗҲе№¶commit

$ git pull --rebase origin master

еҗҲе№¶ж—¶еҸӘдә§з”ҹеҗҲе№¶жҸҗдәӨвҖӢ

иҝҳжңүе…¶е®ғиҜҙжі•пјҡеҗҲе№¶ж—¶дёҚеҗҲе№¶еҺҶеҸІпјӣеҗҲе№¶ж—¶е°ҶжүҖжңүcommitеҗҲдёәдёҖдёӘгҖӮ

git pull --squash another

дёҖиҲ¬з”ЁдәҺmasterеҲҶж”ҜпјҢеғҸlinusзҡ„linuxеҶ…ж ёдёҖж ·гҖӮ жӣҙеӨҡдҝЎжҒҜи§ҒгҖҠGit merge no history commitгҖӢгҖҠgit merge вҖ“squashд»Ӣз»ҚгҖӢ

ж·»еҠ з©әзӣ®еҪ•вҖӢ

$ find . -type d -empty -exec touch {}/.gitkeep \;

жү§иЎҢд»ҘдёҠе‘Ҫд»Өе°ҶдјҡеңЁз©әзӣ®еҪ•дёӯж·»еҠ дёҖдёӘ.gitkeepж–Үд»¶пјҢиҝҷж ·е°ұеҸҜд»ҘдҝқиҜҒз©әзӣ®еҪ•дёҚиў«еҝҪз•ҘпјҢдё”дёҚеҪұе“ҚеҺҹд»Јз Ғзҡ„з»“жһ„гҖӮ

д»ҘдёҠеҸӮиҖғиҪ¬иҮӘ Gitж·ұеәҰдҪҝз”Ёз»ҸйӘҢжҖ»з»“

иҫ“еҮәжңҖеҗҺдёҖж¬ЎжҸҗдәӨзҡ„ж”№еҸҳвҖӢ

иҝҷдёӘе‘Ҫд»ӨпјҢжҲ‘з»ҸеёёдҪҝз”Ёе®ғ жқҘеҸ‘йҖҒе…¶д»–жІЎжңүдҪҝз”Ёgitзҡ„дәәжқҘжЈҖжҹҘжҲ–иҖ…йӣҶжҲҗжүҖдҝ®ж”№зҡ„гҖӮе®ғдјҡиҫ“еҮәжңҖиҝ‘жҸҗдәӨзҡ„дҝ®ж”№еҶ…е®№еҲ°дёҖдёӘzipж–Үд»¶дёӯгҖӮ

git archive -o ../updated.zip HEAD $(git diff --name-only HEAD^)

иҫ“еҮәдёӨдёӘжҸҗдәӨй—ҙзҡ„ж”№еҸҳвҖӢ

зұ»дјјзҡ„пјҢеҰӮжһңдҪ йңҖиҰҒиҫ“еҮәжҹҗдёӨдёӘжҸҗдәӨй—ҙзҡ„ж”№еҸҳж—¶пјҢдҪ еҸҜд»ҘдҪҝз”ЁиҝҷдёӘгҖӮ

git archive -o ../latest.zip NEW_COMMIT_ID_HERE $(git diff --name-only OLD_COMMIT_ID_HERE NEW_COMMIT_ID_HERE)

е…ӢйҡҶ жҢҮе®ҡзҡ„иҝңзЁӢеҲҶж”ҜвҖӢ

еҰӮжһңдҪ жёҙжңӣеҸӘе…ӢйҡҶиҝңзЁӢд»“еә“зҡ„дёҖдёӘжҢҮе®ҡеҲҶж”ҜпјҢиҖҢдёҚжҳҜж•ҙдёӘд»“еә“еҲҶж”ҜпјҢиҝҷеҜ№дҪ её®еҠ©еҫҲеӨ§гҖӮ

git init
git remote add -t BRANCH_NAME_HERE -f origin REMOTE_REPO_URL_PATH_HERE
git checkout BRANCH_NAME_HERE

еә”з”Ё д»ҺдёҚзӣёе…ізҡ„жң¬ең°д»“еә“жқҘзҡ„иЎҘдёҒвҖӢ

еҰӮжһңдҪ йңҖиҰҒе…¶е®ғдёҖдәӣдёҚзӣёе…ізҡ„жң¬ең°д»“еә“дҪңдёәдҪ зҺ°еңЁд»“еә“зҡ„иЎҘдёҒпјҢиҝҷйҮҢе°ұжҳҜйҖҡеҫҖйӮЈйҮҢзҡ„жҚ·еҫ„гҖӮ

git --git-dir=PATH_TO_OTHER_REPOSITORY_HERE/.git format-patch -k -1 --stdout COMMIT_HASH_ID_HERE| git am -3 -k

жЈҖжөӢ дҪ зҡ„еҲҶж”Ҝзҡ„ж”№еҸҳжҳҜеҗҰдёәе…¶е®ғеҲҶж”Ҝзҡ„дёҖйғЁеҲҶвҖӢ

cherryе‘Ҫд»Өи®©жҲ‘д»¬жЈҖжөӢдҪ зҡ„еҲҶж”Ҝзҡ„ж”№еҸҳжҳҜеҗҰеҮәзҺ°еңЁе…¶е®ғдёҖдәӣеҲҶж”ҜдёӯгҖӮе®ғйҖҡиҝҮ+жҲ–иҖ…-з¬ҰеҸ·жқҘжҳҫзӨәд»ҺеҪ“еүҚеҲҶж”ҜдёҺжүҖз»ҷзҡ„еҲҶж”Ҝд№Ӣй—ҙзҡ„ж”№еҸҳпјҡжҳҜеҗҰеҗҲе№¶дәҶ(merged)гҖӮ.+ жҢҮзӨәжІЎжңүеҮәзҺ°еңЁжүҖз»ҷеҲҶж”ҜдёӯпјҢеҸҚд№ӢпјҢ- е°ұиЎЁзӨәеҮәзҺ°еңЁдәҶжүҖз»ҷзҡ„еҲҶж”ҜдёӯдәҶгҖӮиҝҷйҮҢе°ұжҳҜеҰӮдҪ•еҺ»жЈҖжөӢпјҡ

git cherry -v OTHER_BRANCH_NAME_HERE
# дҫӢеҰӮ: жЈҖжөӢmasterеҲҶж”Ҝ
git cherry -v master

ејҖе§ӢдёҖдёӘж— еҺҶеҸІзҡ„ж–°еҲҶж”ҜвҖӢ

жңүж—¶пјҢдҪ йңҖиҰҒејҖе§ӢдёҖдёӘж–°еҲҶж”ҜпјҢдҪҶжҳҜеҸҲдёҚжғіжҠҠеҫҲй•ҝеҫҲй•ҝзҡ„еҺҶеҸІи®°еҪ•еёҰиҝӣжқҘпјҢдҫӢеҰӮпјҢдҪ жғіеңЁе…¬дј—еҢәеҹҹпјҲејҖжәҗпјүж”ҫзҪ®дҪ зҡ„д»Јз ҒпјҢдҪҶжҳҜеҸҲдёҚжғіеҲ«дәәзҹҘйҒ“е®ғзҡ„еҺҶеҸІи®°еҪ•гҖӮ

git checkout --orphan NEW_BRANCH_NAME_HERE

ж— еҲҮжҚўеҲҶж”Ҝзҡ„д»Һе…¶е®ғеҲҶж”ҜCheckoutж–Үд»¶вҖӢ

дёҚжғіеҲҮжҚўеҲҶж”ҜпјҢдҪҶжҳҜеҸҲжғід»Һе…¶е®ғеҲҶж”ҜдёӯиҺ·еҫ—дҪ йңҖиҰҒзҡ„ж–Үд»¶пјҡ

git checkout BRANCH_NAME_HERE -- PATH_TO_FILE_IN_BRANCH_HERE

еҝҪз•Ҙе·ІиҝҪиёӘж–Үд»¶зҡ„еҸҳеҠЁвҖӢ

еҰӮжһңжӮЁжӯЈеңЁдёҖдёӘеӣўйҳҹдёӯе·ҘдҪңпјҢиҖҢдё”еӨ§е®¶йғҪеңЁеҗҢдёҖжқЎbranchдёҠйқўе·ҘдҪңпјҢйӮЈд№ҲжӮЁеҫҲжңүеҸҜиғҪдјҡз»Ҹеёёз”ЁеҲ°fetchе’ҢmergeгҖӮдҪҶжҳҜжңүж—¶еҖҷиҝҷж ·дјҡйҮҚзҪ®жӮЁзҡ„зҺҜеўғй…ҚзҪ®ж–Үд»¶пјҢеҰӮжӯӨзҡ„иҜқпјҢжӮЁе°ұеҫ—еңЁжҜҸж¬ЎmergeеҗҺдҝ®ж”№е®ғгҖӮдҪҝз”ЁиҝҷдёҖе‘Ҫд»ӨпјҢжӮЁе°ұиғҪиҰҒжұӮgitеҝҪи§ҶжҢҮе®ҡж–Үд»¶зҡ„еҸҳеҠЁгҖӮиҝҷж ·пјҢдёӢеӣһдҪ еҶҚmergeзҡ„иҜқпјҢиҝҷдёӘж–Үд»¶е°ұдёҚдјҡиў«дҝ®ж”№дәҶгҖӮ

git update-index --assume-unchanged PATH_TO_FILE_HERE

жЈҖжҹҘжҸҗдәӨзҡ„еҸҳеҠЁжҳҜеҗҰжҳҜreleaseзҡ„дёҖйғЁеҲҶвҖӢ

name-revе‘Ҫд»ӨиғҪе‘ҠиҜүжӮЁдёҖдёӘcommitзӣёеҜ№дәҺжңҖиҝ‘дёҖж¬Ўreleaseзҡ„дҪҚзҪ®гҖӮдҪҝз”ЁиҝҷжқЎе‘Ҫд»ӨпјҢжӮЁе°ұеҸҜд»ҘжЈҖжҹҘжӮЁжүҖеҒҡеҮәзҡ„ж”№еҠЁжҳҜеҗҰжҳҜreleaseзҡ„дёҖйғЁеҲҶдәҶгҖӮ

git name-rev --name-only COMMIT_HASH_HERE

дҪҝз”ЁrebaseжҺЁйҖҒиҖҢйқһmergeвҖӢ

еҰӮжһңжӮЁжӯЈеңЁеӣўйҳҹдёӯе·ҘдҪңе№¶дё”ж•ҙдёӘеӣўйҳҹйғҪеңЁеҗҢдёҖжқЎbranchдёҠйқўе·ҘдҪңпјҢйӮЈд№ҲжӮЁе°ұеҫ—з»Ҹеёёең°иҝӣиЎҢfetch/mergeжҲ–иҖ…pullгҖӮGitдёӯпјҢеҲҶж”Ҝзҡ„еҗҲе№¶д»ҘжүҖжҸҗдәӨзҡ„mergeжқҘи®°еҪ•пјҢд»ҘжӯӨиЎЁжҳҺдёҖжқЎfeatureеҲҶж”ҜдҪ•ж—¶дёҺдё»еҲҶж”ҜеҗҲе№¶гҖӮдҪҶжҳҜеңЁеӨҡеӣўйҳҹжҲҗе‘ҳе…ұеҗҢе·ҘдҪңдәҺдёҖжқЎbranchзҡ„жғ…еҪўдёӯпјҢеёёи§„зҡ„mergeдјҡеҜјиҮҙlogдёӯеҮәзҺ°еӨҡжқЎж¶ҲжҒҜпјҢд»ҺиҖҢдә§з”ҹж··ж·ҶгҖӮеӣ жӯӨпјҢжӮЁеҸҜд»ҘеңЁpullзҡ„ж—¶еҖҷдҪҝз”ЁrebaseпјҢд»ҘжӯӨжқҘеҮҸе°‘ж— з”Ёзҡ„mergeж¶ҲжҒҜпјҢд»ҺиҖҢдҝқжҢҒеҺҶеҸІи®°еҪ•зҡ„жё…жҷ°гҖӮ

git pull --rebase

жӮЁд№ҹеҸҜд»Ҙе°ҶжҹҗжқЎbranchй…ҚзҪ®дёәжҖ»жҳҜдҪҝз”ЁrebaseжҺЁйҖҒпјҡ

git config branch.BRANCH_NAME_HERE.rebase true

Git revert еҗҺеҶҚж¬ЎMergeд»Јз ҒдёўеӨұй—®йўҳвҖӢ

и§ЈеҶіGit Revertж“ҚдҪңеҗҺеҶҚж¬ЎMergeд»Јз Ғиў«еҶІжҺүзҡ„й—®йўҳгҖӮ

дә§з”ҹиҝҮзЁӢвҖӢ

  • ејҖеҸ‘еҲҶж”ҜDev й”ҷиҜҜзҡ„PRеҗҲе№¶еҲ°дәҶ зӣ®ж ҮеҲҶж”ҜA
  • дҪҝз”Ё git revert ж’Өй”ҖдәҶиҝҷж¬Ў PR еҗҲе№¶
  • еҗҺйқўжғізңҹжӯЈе°Ҷ еҲҶж”ҜDev еҗҲе№¶еҲ° еҲҶж”ҜA ж—¶пјҢеҚҙеҸ‘зҺ°жІЎжңүз¬¬дёҖж¬ЎPRзҡ„д»Јз Ғ

и§ЈеҶіж–№ејҸ1пјҡвҖӢ

  • еҲ йҷӨзҺ°жңүзҡ„ зӣ®ж ҮеҲҶж”ҜA
  • д»Һ Master йҮҚж–°еҲҮдёҖдёӘзӣ®ж ҮеҲҶж”ҜA
  • е°ҶејҖеҸ‘еҲҶж”ҜDevеҗҲе№¶еҲ°зӣ®ж ҮеҲҶж”ҜA

и§ЈеҶіж–№ејҸ2пјҡвҖӢ

еҜ№ revert зҡ„йӮЈж¬ЎжҸҗдәӨи®°еҪ•еҶҚж¬ЎиҝӣиЎҢrevert (е®ҳж–№жҺЁиҚҗж–№жі•)

git checkout A
git log

# жүҫеҲ° revert зҡ„йӮЈжқЎжҸҗдәӨи®°еҪ•
# жіЁж„Ҹ: revert зӣёе…ізҡ„дјҡжңүдёӨжқЎи®°еҪ•
#  з¬¬дёҖжқЎжҳҜ revertпјҢз¬¬дәҢжқЎжҳҜ revert еҗҺ merge зҡ„и®°еҪ•пјҢиҝҷйҮҢеҸ–з¬¬дёҖжқЎ

# revertд№ӢеүҚзҡ„revertз¬¬дёҖжқЎcommit_id
git revert <commit_id>

git revert е’Ң git resetзҡ„еҢәеҲ«вҖӢ

  • git revertжҳҜз”ЁдёҖж¬Ўж–°зҡ„commitжқҘеӣһж»ҡд№ӢеүҚзҡ„commitпјҢgit resetжҳҜзӣҙжҺҘеҲ йҷӨжҢҮе®ҡзҡ„commitгҖӮ
